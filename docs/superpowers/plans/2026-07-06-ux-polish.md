final class OnboardingTests: XCTestCase {

    func testHasCompletedOnboardingDefaultsToFalse() throws {
        let config = Config.load()
        XCTAssertFalse(config.hasCompletedOnboarding,
            "fresh config should not have completed onboarding")

        var config = Config.load()

        // Config saves via JSONEncoder — verify the field encodes
        let data = try JSONEncoder().encode(config)
        let decoded = try JSONDecoder().decode(Config.self, from: data)
        XCTAssertTrue(decoded.hasCompletedOnboarding)
    }
}

/// A first-launch onboarding window that guides users through

    private var microphoneStatus: NSTextField!
    private var speechStatus: NSTextField!
    private var claudeStatus: NSTextField!

    init(config: Config, onComplete: @escaping () -> Void) { ... }

    func show() { ... }

    // MARK: - Permission checks

    private func checkAccessibility() { ... }
    private func requestMicrophone() { ... }
    private func requestSpeechRecognition() { ... }
    private func loginClaude() { ... }

    // MARK: - Completion

    private func finish() {
        config.hasCompletedOnboarding = true
        try? config.save()
        window?.close()
        onComplete()

Replace the flat vertical stack with grouped cards:

```swift
func makeView() -> NSView {
    // ... container setup ...

    // Section: Appearance
    let appearanceHeader = makeSectionHeader("APPEARANCE")
    let appearanceCard = makeAppearanceCard()

    // Section: AI Backend
    let backendHeader = makeSectionHeader("AI BACKEND")
    let backendCard = makeBackendCard()

    // Section: Privacy & Recording
    let privacyHeader = makeSectionHeader("PRIVACY & RECORDING")
    let privacyCard = makePrivacyCard()

    // Section: Account
    let accountHeader = makeSectionHeader("ACCOUNT")
    let accountCard = makeAccountCard()

    let stack = NSStackView(views: [
        appearanceHeader, appearanceCard,
        backendHeader, backendCard,
        privacyHeader, privacyCard,
        accountHeader, accountCard,
    ])
    // ... layout ...
}

private func makeSectionHeader(_ text: String) -> NSTextField {
    let label = NSTextField(labelWithString: "")
    label.attributedStringValue = Theme.label(text)
    return label
}
```

Each card method creates a `Theme.makeCard()` and populates it with the relevant controls. The existing control wiring (target/action, popups, etc.) stays identical — only the layout changes.
